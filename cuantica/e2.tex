\textbf{1.4} Demostrar que (a) $[A, B] = -[B, A]$,
(b) $[A^m, A^n] = 0, \forall m, n$, (c) [$A^2, B] = A[A,B] + [A,B]A$,
(d) $[a,[B,C]] + [B,[C,A]]+[C,[A,B]]=0$.

\textbf{DEMOSTRACIONES}

(a)
\begin{align*}
    [A,B]f = ABf - BAf = -(BAf-ABf) = -[B,A]f \implies [A,B]=-[B,A]
\end{align*}

(b)
\begin{align*}
    [A^m, A^n]f &= A^m A^n f - A^n A^m f = A^{m + n} f - A{n + m} f
    = A^{m + n} f - A^{m + n} f = 0\\ &\implies [A^m, A^n] = 0
\end{align*}

(c)
\begin{align*}
    [A^2, B]f &= A^2Bf - BA^2 f = AABf - BAAf + ABAF - ABAF\\
    &= A(ABf- BAf) + (AB-BA)Af = A[A,B]f + [B,A]Af\\
    &\implies [A^2, B] = A[A,B] + [B,A]A 
\end{align*}
