\section*{Solución a los ejercicios}

\textbf{1.3} Evaluar los conmutadores (a) $[x, y]$, (b) $[p_x, p_y]$,
(c) $[x, p_x]$, (d) $[x^2, p_x]$, (e) $[x^n, p_x]$.  

\textbf{SOLUCIÓN}
(a) \begin{gather*}
    [x, y]f = xyf - yxf = xyf - xyf \implies [x, y] = 0
\end{gather*}

(b)
\begin{align*}
    [p_x, p_y]f &= p_x p_y f - p_y p_x f
    = \frac{\hbar}{i} \partial_x \left(\frac{\hbar}{i} \partial_y f\right)
    - \frac{\hbar}{i} \partial_y \left(\frac{\hbar}{i} \partial_y f\right)\\
    &= -\hbar \partial_{xy}^2 f + \hbar \partial_{yx}f = 0\\
    &\implies [p_x, p_y] = 0
\end{align*}

Esto ultimo conmuta por el teorema de Clairaut.

(c)
\begin{align*}
    [x, p_x]f &= xp_x f - p_x (x f)
    = x \frac{\hbar}{i}Df - \frac{\hbar}{i}(f + xDf)\\
    &\implies [x, p_x] = i\hbar
\end{align*}

(d)
\begin{align*}
    [x^2, p_x]f &= x^2\frac{\hbar}{i}Df - \frac{\hbar}{i}D(x^2f)
    = x^2 \frac{\hbar}{i}Df - \frac{\hbar}{i}(2xf + x^2Df)\\
    &\implies [x^2, p_x] = 2 ix\hbar
\end{align*}

(e)
\begin{align*}
    [x^n, p_x]f &= x^n\frac{\hbar}{i}Df - \frac{\hbar}{i}(nx^{n - 1} + x^n Df)
    \implies [x^n, p_x] = ni\hbar x^{n-1}
\end{align*}
