    \begin{gather*}
        S_x = \sum x_i, \quad
        S_y = \sum y_i\\
        S_{xx} = \sum x_i^2, \quad
        S_{xy} = \sum x_i y_i\\
        b = \frac{
            n S_{xy} - S_x S_y
        }{nS_{xx} - S_x^2}, \quad
        a = \frac{S_y - bS_x}{n}\\
        y = a + bx\\
r=\frac{n S_{xy}-S_x S_y}{\sqrt{\big(n S_{xx}-S_x^2\big)\big(n S_{yy}-S_y^2\big)}}
    \end{gather*}
